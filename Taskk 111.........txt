#include <iostream>
using namespace std;
int main() {
    char str[100];
    cin >> str;
    int i = 0, flag = 0;
    while (str[i] != '\0') {
        if (str[i] < '0' || str[i] > '9') {
            flag = 1;
            break;
        }
        i++;
    }
    if (flag == 0)
        cout << "Numeric constant";
    else
        cout << "Not numeric";
    return 0;
}

#include <iostream>
using namespace std;
int main() {
    char exp[100];
    cin >> exp;
    char op[] = {'+', '-', '*', '/', '%', '=', '\0'};
    int i = 0, count = 1;
    while (exp[i] != '\0') {
        for (int j = 0; op[j] != '\0'; j++) {
            if (exp[i] == op[j]) {
                cout << "operator" << count << ":  " << exp[i] << endl;
                count++;
            }
        }
        i++;
    }
    return 0;
}
